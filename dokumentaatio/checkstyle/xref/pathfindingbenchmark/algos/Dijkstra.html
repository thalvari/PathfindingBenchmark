6   package pathfindingbenchmark.algos;
7   
8   import java.math.BigDecimal;
9   import java.math.MathContext;
10  import java.math.RoundingMode;
11  import java.util.ArrayList;
12  import java.util.List;
13  import java.util.PriorityQueue;
14  import pathfindingbenchmark.grid.Grid;
15  import pathfindingbenchmark.grid.Node;
16  
17  /**
18   * Dijkstran algoritmin toteutus.
19   *
20   * @author thalvari
21   */
22  public class Dijkstra {
23  
24      private final Grid grid;
25      private final double cost[];
26      private final Node path[];
27      private final boolean visited[];
28  
29      /**
30       * Konstruktori luo aputietorakenteet.
31       *
32       * @param grid Verkko.
33       */
34      public Dijkstra(Grid grid) {
35          this.grid = grid;
36          cost = new double[grid.getN() + 1];
37          path = new Node[grid.getN() + 1];
38          visited = new boolean[grid.getN() + 1];
39      }
40  
41      /**
42       * Itse algoritmi laskee jokaisen solmun etäisyyden lähtösolmuun.
43       *
44       * @param s Lähtösolmu.
45       */
46      public void run(Node s) {
47          init(s);
48          PriorityQueue<Node> heap = new PriorityQueue();
49          heap.add(new Node(s.getIdx(), 0.0, grid));
50          while (!heap.isEmpty()) {
51              Node u = heap.poll();
52              if (visited[u.getIdx()]) {
53                  continue;
54              }
55  
56              visited[u.getIdx()] = true;
57              for (Node v : grid.getAdjList(u)) {
58                  if (!visited[v.getIdx()]) {
59                      relax(u, v);
60                      heap.add(new Node(v.getIdx(), cost[v.getIdx()], grid));
61                  }
62              }
63          }
64      }
65  
66      private void init(Node s) {
67          for (int i = 1; i <= grid.getN(); i++) {
68              visited[i] = false;
69              cost[i] = Double.MAX_VALUE;
70              path[i] = null;
71          }
72  
73          cost[s.getIdx()] = 0;
74      }
75  
76      private void relax(Node u, Node v) {
77          if (cost[v.getIdx()] > cost[u.getIdx()] + v.getCost()) {
78              cost[v.getIdx()] = cost[u.getIdx()] + v.getCost();
79              path[v.getIdx()] = u;
80          }
81      }
82  
83      /**
84       * Palauttaa solmun etäisyyden lähtösolmuun.
85       *
86       * @param t Solmu.
87       * @return Etäisyys.
88       */
89      public double getCost(Node t) {
90          return cost[t.getIdx()];
91      }
92  
93      /**
94       * Palauttaa solmun etäisyyden lähtösolmuun pyöristettynä kuten
95       * scenario-tiedostoissa.
96       *
97       * @param t Solmu.
98       * @return Etäisyys.
99       */
100     public String getRoundedCost(Node t) {
101         return new BigDecimal(getCost(t))
102                 .round(new MathContext(6, RoundingMode.HALF_EVEN))
103                 .stripTrailingZeros()
104                 .toString();
105     }
106 
107     /**
108      * Tulostaa verkon, jossa lyhin polku annettuun solmuun on merkitty.
109      *
110      * @param t Solmu.
111      */
112     public void printShortestPath(Node t) {
113         Grid tempGrid = grid.cloneGrid();
114         List<Node> nodesInPath = getShortestPath(t);
115 //        tempGrid.markVisited(visited);
116         tempGrid.markPath(nodesInPath);
117         tempGrid.printGrid();
118     }
119 
120     private List<Node> getShortestPath(Node t) {
121         List<Node> nodesInPath = new ArrayList();
122         nodesInPath.add(t);
123         Node u = path[t.getIdx()];
124         while (u != null) {
125             nodesInPath.add(u);
126             u = path[u.getIdx()];
127         }
128 
129         return nodesInPath;
130     }
131 }
