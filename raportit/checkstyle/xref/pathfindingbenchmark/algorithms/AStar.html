11   * Algoritmin A* toteutus.
12   *

15  public class AStar extends AStarAbstract {
16  
17      /**
18       * Konstruktori k채ytt채채 yl채luokan konstruktoria.
19       *
20       * @param grid Verkko.
21       */
22      public AStar(Grid grid) {
23          super(grid);
24      }
25  
26      @Override
27      protected double heuristic(int idx) {
28          int xDif = Math.abs(grid.getX(idx) - grid.getX(goalIdx));
29          int yDif = Math.abs(grid.getY(idx) - grid.getY(goalIdx));
30          return Math.max(xDif, yDif) + (Math.sqrt(2) - 1) * Math.min(xDif, yDif);
31      }
32  }
