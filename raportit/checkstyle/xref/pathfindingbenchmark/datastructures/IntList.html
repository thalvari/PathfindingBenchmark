6   package pathfindingbenchmark.datastructures;
7   
8   /**
9    * Kokonaislukulistan toteutus.
10   *
11   * @author thalvari
12   */
13  public class IntList {
14  
15      private static final int INIT_ARR_LEN = 8;
16      private int[] arr;
17      private int length;
18  
19      /**
20       * Konstruktori.
21       */
22      public IntList() {
23          arr = new int[INIT_ARR_LEN];
24      }
25  
26      /**
27       * Lisää kokonaisluvun listalle.
28       *
29       * @param i Kokonaisluku.
30       */
31      public void add(int i) {
32          checkIfFull();
33          arr[length] = i;
34          length++;
35      }
36  
37      private void checkIfFull() {
38          if (length == arr.length) {
39              int[] newArr = new int[arr.length * 2];
40              for (int j = 0; j < length; j++) {
41                  newArr[j] = arr[j];
42              }
43  
44              arr = newArr;
45          }
46      }
47  
48      /**
49       * Palauttaa indeksiä vastaavan kokonaisluvun kokonaisluvun.
50       *
51       * @param idx Indeksi.
52       * @return Kokonaisluku.
53       */
54      public int get(int idx) {
55          return arr[idx];
56      }
57  
58      /**
59       * Palauttaa listan koon.
60       *
61       * @return Koko.
62       */
63      public int size() {
64          return length;
65      }
66  }
