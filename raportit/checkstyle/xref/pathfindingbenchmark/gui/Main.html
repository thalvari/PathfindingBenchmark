12  import pathfindingbenchmark.algorithms.AStar;
13  import pathfindingbenchmark.algorithms.AStarAbstract;

22  public class Main {

24      private static final String[] ALGOS = {"Dijkstra", "AStar", "JPS"};
25      private static final int SAMPLE_SIZE = 20;
26      private static final Runtime RUNTIME = Runtime.getRuntime();
27      private static final ThreadMXBean BEAN =
28              ManagementFactory.getThreadMXBean();
29  
30      private static long cpuTimeSum;
31      private static long usedMemorySum;
32  
33      public static void main(String[] args) {
34  //        test("AR0011SR", 65, 84, 203, 71, true);
35  //        test("lak505d", 171, 152, 135, 178, true);
36  //        test("rmtst01", 176, 22, 1, 23, true);
37  
38          test("maze512-1-0", 497, 89, 467, 44, false);
39          test("maze512-32-0", 59, 434, 101, 194, false);
40          test("random512-10-0", 19, 44, 509, 436, false);
41          test("random512-40-5", 114, 240, 82, 322, false);
42          test("8room_000", 7, 463, 484, 37, false);
43          test("64room_000", 496, 505, 48, 17, false);
44      }
45  
46      private static void test(String mapName, int startX, int startY, int goalX,
47              int goalY, boolean print) {
48  
49          Grid grid = new Grid(mapName);
50          printMapInfo(mapName, startX, startY, goalX, goalY, grid);
51          AStarAbstract algo;
52          for (String algoName : ALGOS) {
53              if (algoName.equals("Dijkstra")) {
54                  algo = new Dijkstra(grid);
55              } else if (algoName.equals("AStar")) {
56                  algo = new AStar(grid);
57              } else {
58                  algo = new JPS(grid);
59              }
60  
61              cpuTimeSum = 0;
62              usedMemorySum = 0;
63              runAlgo(algo, startX, startY, goalX, goalY);
64              printStatistics(algo, grid);
65              if (print) {
66                  printMarkedGrid(algo);
67              }
68  
69              System.out.println("");
70          }
71      }
72  
73      private static void printMapInfo(String mapName, int startX, int startY,
74              int goalX, int goalY, Grid grid) {
75          for (int i = 0; i < mapName.length(); i++) {
76              System.out.print("-");
77          }
78  
79          System.out.println("");
80          System.out.println("Kartan nimi: " + mapName);
81          System.out.println("Kartan koko: " + grid.getSize());
82          System.out.println("Solmujen määrä: "
83                  + grid.getPassableNodeCount());
84  
85          System.out.println("Lähtösolmu: (" + startX + ", " + startY + ")");
86          System.out.println("Maalisolmu: (" + goalX + ", " + goalY + ")");
87          for (int i = 0; i < mapName.length(); i++) {
88              System.out.print("-");
89          }
90  
91          System.out.println("\n");
92      }
93  
94      private static void runAlgo(AStarAbstract algo, int startX, int startY,
95              int goalX, int goalY) {
96  
97          for (int i = 0; i < SAMPLE_SIZE; i++) {
98              RUNTIME.gc();
99              long startUsedMemory = RUNTIME.totalMemory() - RUNTIME.freeMemory();
100             long startCpuTime = BEAN.getCurrentThreadCpuTime();
101             algo.run(startX, startY, goalX, goalY);
102             cpuTimeSum += BEAN.getCurrentThreadCpuTime() - startCpuTime;
103             usedMemorySum += RUNTIME.totalMemory() - RUNTIME.freeMemory()
104                     - startUsedMemory;
105         }
106     }
107 
108     private static void printMarkedGrid(AStarAbstract algo) {
109         System.out.println("");
110         String[][] mapData = algo.getMarkedMap();
111         for (int y = 0; y < mapData.length; y++) {
112             for (int x = 0; x < mapData[0].length; x++) {
113                 System.out.print(mapData[y][x]);
114             }
115             System.out.println();
116         }
117     }
118 
119     private static void printStatistics(AStarAbstract algo, Grid grid) {
120         if (algo instanceof Dijkstra) {
121             System.out.println("Dijkstra:");
122         } else if (algo instanceof AStar) {
123             System.out.println("A*:");
124         } else if (algo instanceof JPS) {
125             System.out.println("JPS:");
126         }
127 
128         System.out.println("Lyhimmän polun pituus: " + algo.getRoundedDist(6)
129                 + ".");
130 
131         System.out.println("Seuraajalistan koon ka: "
132                 + divideAndRound(algo.getSuccListTotalSize(),
133                         algo.getHeapDelMinOperCount() - 1)
134                 + ".");
135 
136         System.out.println("Keon insert-operaatioiden määrä: "
137                 + algo.getHeapInsertOperCount() + ".");
138 
139         System.out.println("Keon del-min-operaatioiden määrä: "
140                 + algo.getHeapDelMinOperCount() + ".");
141 
142         System.out.println("Keon dec-key-operaatioiden määrä: "
143                 + algo.getHeapDecKeyOperCount() + ".");
144 
145         System.out.println("Suoritusaika: "
146                 + divideAndRound(cpuTimeSum, SAMPLE_SIZE * 1000000L)
147                 + " ms.");
148 
149         System.out.println("Käytetty muisti: "
150                 + divideAndRound(usedMemorySum, SAMPLE_SIZE * 1024L * 1024L)
151                 + " MB.");
152     }
153 
154     private static String divideAndRound(long numer, long denom) {
155         if (denom == 0) {
156             return "0";
157         }
158 
159         return new BigDecimal(numer)
160                 .divide(BigDecimal.valueOf(denom), 3, RoundingMode.HALF_UP)
161                 .toString();
162     }
