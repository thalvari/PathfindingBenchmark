19      private static final int DIJKSTRA = 1;
20      private static final int ASTAR = 2;

23  //        run("arena", 2, 2, 34, 46, ASTAR, 30000, false);

29      private static void run(String mapName, int startX, int startY, int goalX,
30              int goalY, int algoIdx, int n, boolean print) {
31  
32          Grid grid = new Grid(mapName);
33          int startIdx = grid.getIdx(startX, startY);
34          int goalIdx = grid.getIdx(goalX, goalY);
35          AStarAbstract algo;
36          switch (algoIdx) {
37              case DIJKSTRA:
38                  algo = new Dijkstra(grid);
39                  break;
40              case ASTAR:
41                  algo = new AStar(grid);
42                  break;
43              default:
44                  return;
45          }
46  
47          for (int i = 0; i < n; i++) {
48              algo.run(startIdx, goalIdx);
49          }
50  
51          if (print) {
52              printMarkedGrid(algo);
53          }
54  
55          printStatistics(algo, grid);
56      }
57  
58      private static void printMarkedGrid(AStarAbstract algo) {
59          String[][] mapData = algo.getMarkedGrid();
60          for (int y = 0; y < mapData.length; y++) {
61              for (int x = 0; x < mapData[0].length; x++) {
62                  System.out.print(mapData[y][x]);
63              }
64              System.out.println();
65          }
66      }
67  
68      private static void printStatistics(AStarAbstract algo, Grid grid) {
69          if (algo instanceof Dijkstra) {
70              System.out.println("Dijkstra:");
71          } else if (algo instanceof AStar) {
72              System.out.println("A*:");
73          }
74  
75          System.out.println("Lyhimmän polun pituus: " + algo.getRoundedDist(6)
76                  + ".");
77          System.out.println("Solmujen määrä: " + grid.getN() + ".");
78          System.out.println("Käsiteltyjen solmujen määrä: "
79                  + algo.getClosedNodeCount() + ".");
80          System.out.println("Keko-operaatioiden määrä: "
81                  + algo.getHeapOperCount() + ".");
82      }
