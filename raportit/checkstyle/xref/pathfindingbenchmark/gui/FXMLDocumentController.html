6   package pathfindingbenchmark.gui;
7   
8   import java.net.URL;
9   import java.util.ResourceBundle;
10  import javafx.event.Event;
11  import javafx.fxml.FXML;
12  import javafx.fxml.Initializable;
13  import javafx.scene.control.ComboBox;
14  import javafx.scene.control.Label;
15  import javafx.scene.control.RadioButton;
16  import javafx.scene.control.TextField;
17  import javafx.scene.control.ToggleGroup;
18  import javafx.scene.image.ImageView;
19  import javafx.scene.layout.BorderPane;
20  import pathfindingbenchmark.util.MapReader;
21  import pathfindingbenchmark.wrapper.Wrapper;
22  
23  /**
24   *
25   * @author thalvari
26   */
27  public class FXMLDocumentController implements Initializable {
28  
29      @FXML
30      private ToggleGroup algoToggleGroup;
31      @FXML
32      private Label avgSuccListSizeLabel;
33      @FXML
34      private Label closedNodeLabel;
35      @FXML
36      private Label cpuTimeLabel;
37      @FXML
38      private Label errorLabel;
39      @FXML
40      private TextField goalXTextField;
41      @FXML
42      private TextField goalYTextField;
43      @FXML
44      private Label heapOperLabel;
45      @FXML
46      private ComboBox mapComboBox;
47      @FXML
48      private ImageView mapImageView;
49      @FXML
50      private Label mapSizeLabel;
51      @FXML
52      private Label maxHeapSizeLabel;
53      @FXML
54      private Label nodeLabel;
55      @FXML
56      private Label pathLenLabel;
57      @FXML
58      private BorderPane root;
59      @FXML
60      private TextField startXTextField;
61      @FXML
62      private TextField startYTextField;
63      @FXML
64      private Label usedMemoryLabel;
65      private final Wrapper wrapper;
66  
67      public FXMLDocumentController() {
68          wrapper = new Wrapper();
69      }
70  
71      @Override
72      public void initialize(URL location, ResourceBundle resources) {
73          if (MapReader.MAP_NAMES != null) {
74              for (int i = 0; i < MapReader.MAP_NAMES.size(); i++) {
75                  mapComboBox.getItems().add(MapReader.MAP_NAMES.get(i));
76              }
77          } else {
78              errorLabel.setText("No maps");
79          }
80      }
81  
82      @FXML
83      private void handleFindButton(Event event) {
84          int startX;
85          int startY;
86          int goalX;
87          int goalY;
88          try {
89              startX = Integer.parseInt(startXTextField.getText());
90              startY = Integer.parseInt(startYTextField.getText());
91              goalX = Integer.parseInt(goalXTextField.getText());
92              goalY = Integer.parseInt(goalYTextField.getText());
93          } catch (NumberFormatException e) {
94              errorLabel.setText("Out of bounds");
95              return;
96          }
97  
98          if (!wrapper.checkStartGoal(startX, startY, goalX, goalY)) {
99              errorLabel.setText("Out of bounds");
100             return;
101         }
102 
103         errorLabel.setText("");
104         wrapper.setAlgo(((RadioButton) algoToggleGroup.getSelectedToggle())
105                 .getText());
106 
107         wrapper.runAlgo(startX, startY, goalX, goalY);
108         showResults();
109         root.getScene().getWindow().sizeToScene();
110     }
111 
112     @FXML
113     private void handleMapComboBox(Event event) {
114         wrapper.setGrid((String) mapComboBox.getValue());
115         showNewMap();
116         root.getScene().getWindow().sizeToScene();
117     }
118 
119     private void showNewMap() {
120         avgSuccListSizeLabel.setText("");
121         closedNodeLabel.setText("");
122         cpuTimeLabel.setText("");
123         errorLabel.setText("");
124         goalXTextField.clear();
125         goalYTextField.clear();
126         heapOperLabel.setText("");
127         mapImageView.setImage(wrapper.getMapAsWritableImage());
128         mapSizeLabel.setText(wrapper.getGrid().getWidth() + " x "
129                 + wrapper.getGrid().getHeight());
130 
131         maxHeapSizeLabel.setText("");
132         nodeLabel.setText("" + wrapper.getGrid().getPassableNodeCount());
133         pathLenLabel.setText("");
134         startXTextField.clear();
135         startYTextField.clear();
136         usedMemoryLabel.setText("");
137     }
138 
139     private void showResults() {
140         avgSuccListSizeLabel.setText(wrapper.getAvgSuccListSize());
141         closedNodeLabel.setText("" + wrapper.getClosedNodePercentage());
142         cpuTimeLabel.setText(wrapper.getCpuTime());
143         heapOperLabel.setText("" + wrapper.getAlgo().getHeapOperCount());
144         mapImageView.setImage(wrapper.getMapAsWritableImage());
145         maxHeapSizeLabel.setText("" + wrapper.getAlgo().getMaxHeapSize());
146         pathLenLabel.setText(wrapper.getPathLen());
147         usedMemoryLabel.setText(wrapper.getUsedMemory());
148     }
149 }
