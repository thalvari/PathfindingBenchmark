7   
8   import java.math.BigDecimal;
9   import java.math.MathContext;
10  import java.math.RoundingMode;
11  import java.util.ArrayList;
12  import java.util.List;
13  import pathfindingbenchmark.grid.Grid;
14  import pathfindingbenchmark.grid.Node;
15  
16  /**
17   * Reitinhakualgoritmia kuvaava abstrakti luokka.
18   *
19   * @author thalvari
20   */
21  public abstract class Algo {
22  
23      /**
24       * Verkko, jossa algoritmi ajetaan.
25       */
26      protected final Grid grid;
27  
28      /**
29       * Päivittyvä lista etäisyyksistä lähtösolmuun.
30       */
31      protected final double cost[];
32  
33      /**
34       * Päivittyvä lista edellisistä solmuista polulla.
35       */
36      protected final Node path[];
37  
38      /**
39       * Päivittyvä lista vierailluista solmuista.
40       */
41      protected final boolean visited[];
42  
43      /**
44       * Lähtösolmu.
45       */
46      protected Node s;
47  
48      /**
49       * Maalisolmu.
50       */
51      protected Node t;
52  
53      /**
54       * Konstruktori luo aputietorakenteet.
55       *
56       * @param grid Verkko.
57       */
58      public Algo(Grid grid) {
59          this.grid = grid;
60          cost = new double[grid.getN() + 1];
61          path = new Node[grid.getN() + 1];
62          visited = new boolean[grid.getN() + 1];
63      }
64  
65      /**
66       * Laskee lyhimmän polun pituuden lähtösolmusta maalisolmuun ja ottaa
67       * kyseisen polun talteen.
68       *
69       * @param s Lähtösolmu.
70       * @param t Maalisolmu.
71       */
72      public abstract void run(Node s, Node t);
73  
74      /**
75       * Palauttaa solmun etäisyyden lähtösolmuun pyöristettynä kuten
76       * scenario-tiedostoissa.
77       *
78       * @return Etäisyys.
79       */
80      public String getRoundedCost() {
81          return BigDecimal.valueOf(getCost())
82                  .round(new MathContext(6, RoundingMode.HALF_EVEN))
83                  .stripTrailingZeros()
84                  .toString();
85      }
86  
87      private double getCost() {
88          return cost[t.getIdx()];
89      }
90  
91      /**
92       * Tulostaa verkon, jossa lyhin polku lähtösolmusta maalisolmuun on
93       * merkitty.
94       */
95      public void printShortestPath() {
96          String[][] mapData = grid.getMapData().clone();
97          List<Node> nodesInPath = getShortestPath();
98  //        markVisited(mapData);
99          markPath(mapData, nodesInPath);
100         printMapData(mapData);
101     }
102 
103     private List<Node> getShortestPath() {
104         List<Node> nodesInPath = new ArrayList();
105         nodesInPath.add(t);
106         Node u = path[t.getIdx()];
107         while (u != null) {
108             nodesInPath.add(u);
109             u = path[u.getIdx()];
110         }
111 
112         return nodesInPath;
113     }
114 
115     private void markPath(String[][] mapData, List<Node> nodesInPath) {
116         for (Node u : nodesInPath) {
117             mapData[u.getY()][u.getX()] = "X";
118         }
119     }
120 
121     private void printMapData(String[][] mapData) {
122         for (int y = 0; y < mapData.length; y++) {
123             for (int x = 0; x < mapData[0].length; x++) {
124                 System.out.print(mapData[y][x]);
125             }
126             System.out.println();
127         }
128     }
