6   package pathfindingbenchmark.algo;
7   
8   import java.util.PriorityQueue;
9   import pathfindingbenchmark.grid.Grid;
10  import pathfindingbenchmark.grid.Node;
11  
12  /**
13   * Dijkstran algoritmin toteutus.
14   *
15   * @author thalvari
16   */
17  public class Dijkstra extends Algo {
18  
19      /**
20       * Konstruktori k채ytt채채 yl채luokan konstruktoria.
21       *
22       * @param grid Verkko.
23       */
24      public Dijkstra(Grid grid) {
25          super(grid);
26      }
27  
28      @Override
29      public void run(Node s, Node t) {
30          init(s, t);
31          PriorityQueue<Node> heap = new PriorityQueue();
32          heap.add(new Node(s.getX(), s.getY(), 0, grid));
33          while (!heap.isEmpty()) {
34              Node u = heap.poll();
35              if (visited[u.getIdx()]) {
36                  continue;
37              }
38  
39              visited[u.getIdx()] = true;
40              for (Node v : grid.getAdjList(u)) {
41                  if (!visited[v.getIdx()]) {
42                      relax(u, v);
43                      heap.add(new Node(v.getX(),
44                              v.getY(),
45                              cost[v.getIdx()],
46                              grid));
47                  }
48              }
49          }
50      }
51  
52      private void init(Node s, Node t) {
53          this.s = s;
54          this.t = t;
55          for (int i = 1; i <= grid.getN(); i++) {
56              cost[i] = Double.MAX_VALUE;
57              path[i] = null;
58              visited[i] = false;
59          }
60  
61          cost[s.getIdx()] = 0;
62      }
63  
64      private void relax(Node u, Node v) {
65          if (cost[v.getIdx()] > cost[u.getIdx()] + v.getCost()) {
66              cost[v.getIdx()] = cost[u.getIdx()] + v.getCost();
67              path[v.getIdx()] = u;
68          }
69      }
70  }
