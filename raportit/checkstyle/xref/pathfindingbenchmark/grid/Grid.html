6   package pathfindingbenchmark.grid;
7   
8   import pathfindingbenchmark.datastructures.MyList;
9   import pathfindingbenchmark.util.Direction;
10  import pathfindingbenchmark.util.MapReader;
11  
12  /**
13   * Kartan esitys verkkona.
14   *
15   * @author thalvari
16   */
17  public class Grid {
18  
19      /**
20       * Kahden solmun etäisyys liikuttaessa pysty- tai vaakatasossa.
21       */
22      public static final int HOR_VER_NODE_DIST = 13860;
23  
24      /**
25       * Kahden solmun etäisyys liikuttaessa viistoon. Tämän ja ylläolevan luvun
26       * suhde on noin sqrt(2).
27       */
28      public static final int DIAG_NODE_DIST = 19601;
29  
30      private final MapReader reader;
31      private final int height;
32      private final int width;
33      private final Node[][] nodes;
34      private final int passableNodeCount;
35  
36      /**
37       * Konstruktori käyttää kartanlukijaa verkon tallentamiseen.
38       *
39       * @param mapName Kartan nimi.
40       */
41      public Grid(String mapName) {
42          reader = new MapReader(mapName);
43          height = reader.getHeight();
44          width = reader.getWidth();
45          nodes = reader.getNodes();
46          passableNodeCount = reader.getPassableNodeCount();
47      }
48  
49      /**
50       * Palauttaa listan naapurisolmuista.
51       *
52       * @param node Solmu.
53       * @return Solmulista.
54       */
55      public MyList<Node> createAdjList(Node node) {
56          MyList<Node> adjList = new MyList<>();
57          for (int y = -1; y <= 1; y++) {
58              for (int x = -1; x <= 1; x++) {
59                  Direction dir = new Direction(x, y);
60                  if ((x != 0 || y != 0)
61                          && ((dir.isDiag()
62                          && isNodeInDirAdj(node, dir))
63                          || (!dir.isDiag()
64                          && isNodeInDirPassable(node, dir)))) {
65  
66                      adjList.add(getNode(node.getX() + x, node.getY() + y));
67                  }
68              }
69          }
70  
71          return adjList;
72      }
73  
74      /**
75       * Tarkastaa voidaanko annetussa suunnassa olevaan solmuun siirtyä.
76       *
77       * @param node Solmu.
78       * @param dir Suunta.
79       * @return Totuusarvo.
80       */
81      public boolean isNodeInDirAdj(Node node, Direction dir) {
82          if (!isInBounds(node.getX() + dir.getX(), node.getY() + dir.getY())) {
83              return false;
84          }
85  
86          return getNodeInDir(node, dir).isPassable()
87                  && (!dir.isDiag()
88                  || (getNode(node.getX(), node.getY() + dir.getY()).isPassable()
89                  && getNode(node.getX() + dir.getX(), node.getY())
90                  .isPassable()));
91      }
92  
93      /**
94       * Tarkastaa sijaitsevatko annetut koordinaatit kartalla.
95       *
96       * @param x X-koordinaatti.
97       * @param y Y-koordinaatti.
98       * @return Totuusarvo.
99       */
100     public boolean isInBounds(int x, int y) {
101         return x >= 0 && x < width && y >= 0 && y < height;
102     }
103 
104     /**
105      * Palauttaa annetussa suunnassa olevan solmun.
106      *
107      * @param node Solmu.
108      * @param dir Suunta.
109      * @return Totuusarvo.
110      */
111     public Node getNodeInDir(Node node, Direction dir) {
112         return getNode(node.getX() + dir.getX(), node.getY() + dir.getY());
113     }
114 
115     /**
116      * Kertoo onko annetussa suunnassa oleva solmu läpikuljettavissa.
117      *
118      * @param node Solmu.
119      * @param dir Suunta.
120      * @return Totuusarvo.
121      */
122     public boolean isNodeInDirPassable(Node node, Direction dir) {
123         if (!isInBounds(node.getX() + dir.getX(), node.getY() + dir.getY())) {
124             return false;
125         }
126 
127         return getNodeInDir(node, dir).isPassable();
128     }
129 
130     /**
131      * Palauttaa solmujen oktiilisen etäisyyden.
132      *
133      * @param node1 Solmu.
134      * @param node2 Toinen solmu.
135      * @return Etäisyys.
136      */
137     public int getNodeDist(Node node1, Node node2) {
138         int xDif = Math.abs(node1.getX() - node2.getX());
139         int yDif = Math.abs(node1.getY() - node2.getY());
140         return HOR_VER_NODE_DIST * Math.max(xDif, yDif)
141                 + (DIAG_NODE_DIST - HOR_VER_NODE_DIST) * Math.min(xDif, yDif);
142     }
143 
144     /**
145      * Palauttaa läpikuljettavien solmujen määrän kartalla.
146      *
147      * @return Määrä.
148      */
149     public int getPassableNodeCount() {
150         return passableNodeCount;
151     }
152 
153     /**
154      * Palauttaa taulukkoesityksen kartasta, johon lyhin polku ja käsitellyt
155      * solmut on merkitty.
156      *
157      * @param goal Maalisolmu.
158      * @return Taulukkoesitys verkosta.
159      */
160     public char[][] getMarkedMap(Node goal) {
161         char[][] markedMap = new char[height][width];
162         markPath(markedMap, goal);
163         for (int y = 0; y < height; y++) {
164             for (int x = 0; x < width; x++) {
165                 if (markedMap[y][x] != 'X' && getNode(x, y).isClosed()) {
166                     markedMap[y][x] = 'o';
167                 } else if (markedMap[y][x] != 'X') {
168                     markedMap[y][x] = getNode(x, y).getSymbol();
169                 }
170             }
171         }
172 
173         return markedMap;
174     }
175 
176     private void markPath(char[][] markedMap, Node goal) {
177         Node prev = goal;
178         while (prev != null) {
179             markedMap[prev.getY()][prev.getX()] = 'X';
180             prev = prev.getPrev();
181         }
182     }

190         return height;
191     }
192 
193     /**
194      * Palauttaa kartan leveyden.
195      *
196      * @return Leveys.
197      */
198     public int getWidth() {
199         return width;
200     }
201 
202     /**
203      * Palauttaa tietyissä koordinaateissa sijaitsevan solmun.
204      *
205      * @param x X-koordinaatti.
206      * @param y Y-koordinaatti.
207      * @return Solmu.
208      */
209     public Node getNode(int x, int y) {
210         return nodes[y][x];
211     }
212 
213     /**
214      * Alustaa kaikki verkon solmut.
215      */
216     public void initNodes() {
217         for (int y = 0; y < height; y++) {
218             for (int x = 0; x < width; x++) {
219                 getNode(x, y).reset();
220             }
221         }
222     }
